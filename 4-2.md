
## Node Class
```java
public class Node {
    private Node node;
    private String string;

    public Node(){
    }

    public Node getNode(){
        return this.node;
    }

    private Node(String string) {
        this.node = null;
        this.string = string;
    }

    public void setAddress(Node node){
        this.node = node;
    }

    public static Node newNode(String string){
        return new Node(string);
    }

    public String getValue(){
        return String.valueOf(this.string);
    }

}
```
